### 1. **Python (Read all files in a directory):**
```python
import os

def read_all_files(directory):
    for root, _, files in os.walk(directory):
        for file in files:
            file_path = os.path.join(root, file)
            try:
                with open(file_path, 'r') as f:
                    content = f.read()
                    print(f"--- {file_path} ---")
                    print(content[:200])  # Print first 200 chars to avoid clutter
            except Exception as e:
                print(f"Error reading {file_path}: {e}")
```

### 2. **JavaScript (Node.js):**
```javascript
const fs = require('fs');
const path = require('path');

function readAllFiles(dir) {
    fs.readdirSync(dir).forEach(file => {
        const fullPath = path.join(dir, file);
        if (fs.lstatSync(fullPath).isDirectory()) {
            readAllFiles(fullPath); // Recursive for directories
        } else {
            try {
                const content = fs.readFileSync(fullPath, 'utf8');
                console.log(`--- ${fullPath} ---\n${content.substring(0, 200)}`);
            } catch (e) {
                console.error(`Error reading ${fullPath}:`, e.message);
            }
        }
    });
}

// Usage: Replace '.' with your project path
readAllFiles('.');
```
